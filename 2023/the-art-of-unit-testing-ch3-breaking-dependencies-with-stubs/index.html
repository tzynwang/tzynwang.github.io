    if ([6, 0].includes(this.timeProvider.getDay())) {
      throw new Error("It's the weekend!");
    }
    const errors: string[] = [];
    this.rules.forEach((rule) => {
      const result = rule(input);
      if (!result.passed) {
        errors.push(`error ${result.reason}`);
      }
    });
    return errors;
  }
}
