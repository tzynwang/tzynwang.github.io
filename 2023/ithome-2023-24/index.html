export function groupPostByYear(posts: MarkdownInstance<Post>[]) {
  const yearsMap: Record<string, MarkdownInstance<Post>[]> = {};
  posts.forEach((post) => {
    const y = dayjs(post.frontmatter.date).format('YYYY');
    if (!yearsMap[y]) {
      yearsMap[y] = [];
    }
    yearsMap[y].push(post);
  });
  return yearsMap;
}
